class Solution:
    def largestOddNumber(self, num: str) -> str:
        for i in range(len(num)-1,-1,-1):
            if i==0 and int(num[i])%2!=0:
                return num[0]
            elif i==0 and int(num[i])%2==0:
                return ''
            elif int(num[i])%2!=0:
                return num[:i+1]
